Example add Material in app

Follow these steps to install Material:

  1. Command in console app: ng add @angular/material

  2. add in app.module.ts this code:

    
import { BrowserAnimationsModule } from '@angular/platform-browser/animations';
                    ...

    @NgModule({
    ...
    imports: [BrowserAnimationsModule],
    ...
    })

  3. Import the component modules

    
import {MatButtonModule} from '@angular/material/button';
import {MatCheckboxModule} from '@angular/material/checkbox';

@NgModule({
  ...
  imports: [MatButtonModule, MatCheckboxModule],
  ...
})
export class PizzaPartyAppModule { }

  4. Include a theme

    in file style.scss include example:
    @import "~@angular/material/prebuilt-themes/indigo-pink.css";
